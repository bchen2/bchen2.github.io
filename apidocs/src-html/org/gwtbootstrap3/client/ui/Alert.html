023import com.google.gwt.dom.client.Element;
024import com.google.gwt.safehtml.shared.SafeHtml;
025import com.google.gwt.user.client.Event;
026import com.google.web.bindery.event.shared.HandlerRegistration;
027import org.gwtbootstrap3.client.shared.event.AlertCloseEvent;
028import org.gwtbootstrap3.client.shared.event.AlertCloseHandler;
029import org.gwtbootstrap3.client.shared.event.AlertClosedEvent;
030import org.gwtbootstrap3.client.shared.event.AlertClosedHandler;
031import org.gwtbootstrap3.client.ui.base.button.CloseButton;
032import org.gwtbootstrap3.client.ui.base.helper.StyleHelper;
033import org.gwtbootstrap3.client.ui.constants.AlertType;
034import org.gwtbootstrap3.client.ui.constants.ButtonDismiss;
035import org.gwtbootstrap3.client.ui.constants.Styles;
036
037/**
038 * Alert block.
039 * <p/>
040 * Use {@link #setDismissable(boolean)} to add a close ("x") button.
041 *
042 * @author Sven Jacobs
043 * @author Joshua Godi
044 * @see org.gwtbootstrap3.client.shared.event.AlertCloseEvent
045 * @see org.gwtbootstrap3.client.shared.event.AlertClosedEvent
046 */
047public class Alert extends HTMLPanel implements HasType<AlertType>, HasResponsiveness {
048    private static final String CLOSE = "close";
049
050    private final CloseButton closeButton = new CloseButton();
051
052    public Alert(final String html) {
053        super(html);
054
055        setStyleName(Styles.ALERT);
056        setType(AlertType.WARNING);
057
058        closeButton.setDismiss(ButtonDismiss.ALERT);
059    }
060
061    public Alert(final String html, final AlertType type) {
062        this(html);
063        setType(type);
064    }
065
066    public Alert(final SafeHtml safeHtml) {
067        this(safeHtml.asString());
068    }
069
070    public Alert(final SafeHtml safeHtml, final AlertType type) {
071        this(safeHtml.asString(), type);
072    }
073
074    @Override
075    protected void onLoad() {
076        super.onLoad();
077
078        // Bind the events
079        bindJavaScriptEvents(getElement());
080    }
081
082    @Override
083    protected void onUnload() {
084        super.onUnload();
085
086        // Unbind the events
087        unbindJavaScriptEvents(getElement());
088    }
089
090    /**
091     * Sets alert type.
092     *
093     * @param type Alert type
094     * @see AlertType
095     */
096    @Override
097    public void setType(final AlertType type) {
098        StyleHelper.addUniqueEnumStyleName(this, AlertType.class, type);
099    }
100
101    @Override
102    public AlertType getType() {
103        return AlertType.fromStyleName(getStyleName());
104    }
105
106    /**
107     * Adds a close button to the alert
108     *
109     * @param dismissable Adds close button when {@code true}
110     */
111    public void setDismissable(final boolean dismissable) {
112        if (dismissable) {
113            insert(closeButton, getElement(), 0, true);
114            addStyleName(Styles.ALERT_DISMISSABLE);
115        } else {
116            closeButton.removeFromParent();
117            removeStyleName(Styles.ALERT_DISMISSABLE);
118        }
119    }
120
121    public boolean isDismissable() {
122        return closeButton.getParent() != null;
123    }
124
125    /**
126     * Closes alert.
127     */
128    public void close() {
129        alert(getElement(), CLOSE);
130    }
131
132    protected void onClose(final Event evt) {
133        fireEvent(new AlertCloseEvent(evt));
134    }
135
136    protected void onClosed(final Event evt) {
137        fireEvent(new AlertClosedEvent(evt));
138    }
139
140    public HandlerRegistration addCloseHandler(final AlertCloseHandler handler) {
141        return addHandler(handler, AlertCloseEvent.getType());
142    }
143
144    public HandlerRegistration addClosedHandler(final AlertClosedHandler handler) {
145        return addHandler(handler, AlertClosedEvent.getType());
146    }
147
148    @Override
149    public void setVisibleOn(final String deviceSizeString) {
150        StyleHelper.setVisibleOn(this, deviceSizeString);
151    }
152
153    @Override
154    public void setHiddenOn(final String deviceSizeString) {
155        StyleHelper.setHiddenOn(this, deviceSizeString);
156    }
157
158    // @formatter:off
159
160    private native void alert(final Element e, final String arg) /*-{
161        $wnd.jQuery(e).alert(arg);
162    }-*/;
163
164    private native void bindJavaScriptEvents(final Element e) /*-{
165        var target = this;
166        var $alert = $wnd.jQuery(e);
167
168        $alert.on('close.bs.alert', function (evt) {
169            target.@org.gwtbootstrap3.client.ui.Alert::onClose(Lcom/google/gwt/user/client/Event;)(evt);
170        });
171
172        $alert.on('closed.bs.alert', function (evt) {
173            target.@org.gwtbootstrap3.client.ui.Alert::onClosed(Lcom/google/gwt/user/client/Event;)(evt);
174        });
175    }-*/;
176
177    private native void unbindJavaScriptEvents(final Element e) /*-{
178        $wnd.jQuery(e).off('close.bs.alert');
179        $wnd.jQuery(e).off('closed.bs.alert');
180    }-*/;
181}
