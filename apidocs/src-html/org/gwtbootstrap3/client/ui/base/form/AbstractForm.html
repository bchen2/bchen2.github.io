028 * @author Sven Jacobs
029 */
030public abstract class AbstractForm extends FormElementContainer {
031    private static final String FORM = "form";
032
033    public AbstractForm() {
034        setElement(Document.get().createFormElement());
035        getElement().setAttribute(Attributes.ROLE, FORM);
036    }
037
038    public String getAction() {
039        return FormElement.as(getElement()).getAction();
040    }
041
042    public void setAction(final String action) {
043        FormElement.as(getElement()).setAction(action);
044    }
045
046    public String getMethod() {
047        return FormElement.as(getElement()).getMethod();
048    }
049
050    public void setMethod(final String method) {
051        FormElement.as(getElement()).setMethod(method);
052    }
053}
