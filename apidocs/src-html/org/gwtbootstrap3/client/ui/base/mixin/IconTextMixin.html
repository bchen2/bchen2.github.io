023import com.google.gwt.user.client.ui.HasText;
024import org.gwtbootstrap3.client.ui.HasIcon;
025import org.gwtbootstrap3.client.ui.HasIconPosition;
026import org.gwtbootstrap3.client.ui.Icon;
027import org.gwtbootstrap3.client.ui.Text;
028import org.gwtbootstrap3.client.ui.base.ComplexWidget;
029import org.gwtbootstrap3.client.ui.constants.*;
030
031/**
032 * Mixin for Widgets that have text and an optional icon.
033 *
034 * @author Sven Jacobs
035 */
036public class IconTextMixin<T extends ComplexWidget & HasText & HasIcon & HasIconPosition> implements
037        HasText, HasIcon, HasIconPosition {
038
039    private final T widget;
040    private final Text text = new Text();
041    private final Text separator = new Text(" ");
042    private Icon icon;
043    private IconPosition iconPosition = IconPosition.LEFT;
044    private IconSize iconSize = IconSize.NONE;
045    private IconFlip iconFlip = IconFlip.NONE;
046    private IconRotate iconRotate = IconRotate.NONE;
047    private boolean iconMuted = false;
048    private boolean iconSpin = false;
049    private boolean iconBordered = false;
050    private boolean iconLight = false;
051
052    public IconTextMixin(final T widget) {
053        this.widget = widget;
054    }
055
056    public void addTextWidgetToParent() {
057        widget.add(text);
058    }
059
060    @Override
061    public void setText(final String text) {
062        this.text.setText(text);
063    }
064
065    @Override
066    public String getText() {
067        return text.getText();
068    }
069
070    @Override
071    public void setIcon(final IconType iconType) {
072        render(new Icon(iconType));
073    }
074
075    @Override
076    public IconType getIcon() {
077        return icon == null ? null : icon.getType();
078    }
079
080    @Override
081    public void setIconPosition(final IconPosition iconPosition) {
082        this.iconPosition = iconPosition;
083        render(icon);
084    }
085
086    @Override
087    public IconPosition getIconPosition() {
088        return iconPosition;
089    }
090
091    @Override
092    public void setIconSize(final IconSize iconSize) {
093        this.iconSize = iconSize;
094        render(icon);
095    }
096
097    @Override
098    public IconSize getIconSize() {
099        return iconSize;
100    }
101
102    @Override
103    public void setIconFlip(IconFlip iconFlip) {
104        this.iconFlip = iconFlip;
105        render(icon);
106    }
107
108    @Override
109    public IconFlip getIconFlip() {
110        return iconFlip;
111    }
112
113    @Override
114    public void setIconRotate(IconRotate iconRotate) {
115        this.iconRotate = iconRotate;
116        render(icon);
117    }
118
119    @Override
120    public IconRotate getIconRotate() {
121        return iconRotate;
122    }
123
124    @Override
125    public void setIconBordered(boolean iconBordered) {
126        this.iconBordered = iconBordered;
127        render(icon);
128    }
129
130    @Override
131    public boolean isIconBordered() {
132        return iconBordered;
133    }
134
135    @Override
136    public void setIconMuted(boolean iconMuted) {
137        this.iconMuted = iconMuted;
138        render(icon);
139    }
140
141    @Override
142    public boolean isIconMuted() {
143        return iconMuted;
144    }
145
146    @Override
147    public void setIconLight(boolean iconLight) {
148        this.iconLight = iconLight;
149        render(icon);
150    }
151
152    @Override
153    public boolean isIconLight() {
154        return iconLight;
155    }
156
157    @Override
158    public void setIconSpin(boolean iconSpin) {
159        this.iconSpin = iconSpin;
160        render(icon);
161    }
162
163    @Override
164    public boolean isIconSpin() {
165        return iconSpin;
166    }
167
168    private void render(final Icon newIcon) {
169        text.removeFromParent();
170        separator.removeFromParent();
171
172        if (icon != null) {
173            icon.removeFromParent();
174        }
175
176        icon = newIcon;
177        icon.setSize(iconSize);
178        icon.setFlip(iconFlip);
179        icon.setRotate(iconRotate);
180        icon.setMuted(iconMuted);
181        icon.setSpin(iconSpin);
182        icon.setBorder(iconBordered);
183        icon.setLight(iconLight);
184
185        if (iconPosition == IconPosition.LEFT) {
186            widget.add(icon);
187            widget.add(separator);
188        }
189
190        widget.add(text);
191
192        if (iconPosition == IconPosition.RIGHT) {
193            widget.add(separator);
194            widget.add(icon);
195        }
196    }
197}
