023import com.google.gwt.dom.client.Element;
024import com.google.gwt.event.logical.shared.ValueChangeHandler;
025import com.google.gwt.event.shared.HandlerRegistration;
026import com.google.gwt.user.client.DOM;
027import com.google.gwt.user.client.ui.HasName;
028import com.google.gwt.user.client.ui.HasValue;
029import org.gwtbootstrap3.client.ui.CheckableInputButton;
030import org.gwtbootstrap3.client.ui.HasActive;
031import org.gwtbootstrap3.client.ui.HasFormValue;
032import org.gwtbootstrap3.client.ui.base.mixin.ActiveMixin;
033import org.gwtbootstrap3.client.ui.constants.ButtonType;
034import org.gwtbootstrap3.client.ui.constants.Styles;
035import org.gwtbootstrap3.client.ui.constants.TypeAttrType;
036
037/**
038 * Base class for {@link org.gwtbootstrap3.client.ui.CheckBoxButton}
039 * and {@link org.gwtbootstrap3.client.ui.RadioButton} which are
040 * encapsulated {@code <input>} elements within {@code <label>}.
041 *
042 * @author Sven Jacobs
043 */
044public abstract class AbstractLabelButton extends AbstractIconButton implements HasActive, HasName, HasFormValue, HasValue<Boolean> {
045
046    private final ActiveMixin<AbstractLabelButton> activeMixin = new ActiveMixin<AbstractLabelButton>(this);
047    protected final CheckableInputButton input;
048
049    protected AbstractLabelButton(final TypeAttrType typeAttr) {
050        super(ButtonType.DEFAULT);
051
052        input = new CheckableInputButton(typeAttr);
053        input.setStyleName("");
054
055        add(input, getElement());
056        iconTextMixin.addTextWidgetToParent();
057    }
058
059    protected AbstractLabelButton(final TypeAttrType typeAttr,
060                                  final String label) {
061
062        this(typeAttr);
063        setText(label);
064    }
065
066    @Override
067    public void setEnabled(final boolean enabled) {
068        if (enabled) {
069            removeStyleName(Styles.DISABLED);
070        } else {
071            addStyleName(Styles.DISABLED);
072        }
073
074        input.setEnabled(enabled);
075    }
076
077    @Override
078    public boolean isEnabled() {
079        return input.isEnabled();
080    }
081
082    @Override
083    public void setActive(final boolean active) {
084        setValue(active);
085    }
086
087    @Override
088    public boolean isActive() {
089        return activeMixin.isActive();
090    }
091
092    @Override
093    public void setName(final String name) {
094        input.setName(name);
095    }
096
097    @Override
098    public String getName() {
099        return input.getName();
100    }
101
102    @Override
103    public String getFormValue() {
104        return input.getFormValue();
105    }
106
107    @Override
108    public void setFormValue(final String value) {
109        input.setFormValue(value);
110    }
111
112    @Override
113    public void setValue(final Boolean value) {
114        setValue(value, false);
115    }
116
117    @Override
118    public void setValue(final Boolean value, final boolean fireEvents) {
119        activeMixin.setActive(value);
120        input.setValue(value, fireEvents);
121    }
122
123    @Override
124    public Boolean getValue() {
125        return input.getValue();
126    }
127
128    @Override
129    public HandlerRegistration addValueChangeHandler(final ValueChangeHandler<Boolean> handler) {
130        return input.addValueChangeHandler(handler);
131    }
132
133    @Override
134    protected Element createElement() {
135        return DOM.createLabel();
136    }
137}
