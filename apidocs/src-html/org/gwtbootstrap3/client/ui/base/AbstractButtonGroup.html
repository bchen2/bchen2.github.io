023import com.google.gwt.user.client.ui.HasName;
024import com.google.gwt.user.client.ui.Widget;
025import org.gwtbootstrap3.client.ui.*;
026import org.gwtbootstrap3.client.ui.base.helper.StyleHelper;
027import org.gwtbootstrap3.client.ui.base.mixin.PullMixin;
028import org.gwtbootstrap3.client.ui.base.mixin.ToggleMixin;
029import org.gwtbootstrap3.client.ui.constants.Pull;
030import org.gwtbootstrap3.client.ui.constants.Styles;
031import org.gwtbootstrap3.client.ui.constants.Toggle;
032
033/**
034 * Abstract base class for button groups.
035 *
036 * @author Sven Jacobs
037 * @author Joshua Godi
038 * @see org.gwtbootstrap3.client.ui.ButtonGroup
039 * @see org.gwtbootstrap3.client.ui.VerticalButtonGroup
040 */
041public abstract class AbstractButtonGroup extends FlowPanel implements HasName, HasToggle, HasJustified, HasPull, HasResponsiveness {
042
043    private final PullMixin<AbstractButtonGroup> pullMixin = new PullMixin<AbstractButtonGroup>(this);
044    private final ToggleMixin<AbstractButtonGroup> toggleMixin = new ToggleMixin<AbstractButtonGroup>(this);
045    private String name;
046
047    protected AbstractButtonGroup(final String styleName) {
048        setStyleName(styleName);
049    }
050
051    /**
052     * Convenience method that will set the name of all child widgets that can have a name
053     *
054     * @param name Name of group
055     * @see #add(com.google.gwt.user.client.ui.Widget)
056     */
057    @Override
058    public void setName(final String name) {
059        this.name = name;
060
061        if (name == null) {
062            return;
063        }
064
065        for (final Widget w : getChildren()) {
066            if (w instanceof HasName) {
067                ((HasName) w).setName(name);
068            }
069        }
070    }
071
072    @Override
073    public String getName() {
074        return this.name;
075    }
076
077    @Override
078    public void setToggle(final Toggle toggle) {
079        toggleMixin.setToggle(toggle);
080    }
081
082    @Override
083    public Toggle getToggle() {
084        return toggleMixin.getToggle();
085    }
086
087    /**
088     * Make a group of buttons stretch at the same size to span the entire width of its parent.
089     * <p/>
090     * <strong>Note:</strong> Justified button groups only work with {@link org.gwtbootstrap3.client.ui.AnchorButton} child elements!
091     *
092     * @param justified Stretch button group
093     */
094    @Override
095    public void setJustified(final boolean justified) {
096        if (justified) {
097            addStyleName(Styles.BTN_GROUP_JUSTIFIED);
098        } else {
099            removeStyleName(Styles.BTN_GROUP_JUSTIFIED);
100        }
101    }
102
103    @Override
104    public boolean isJustified() {
105        return StyleHelper.containsStyle(getStyleName(), Styles.BTN_GROUP_JUSTIFIED);
106    }
107
108    @Override
109    public void setPull(final Pull pull) {
110        pullMixin.setPull(pull);
111    }
112
113    @Override
114    public Pull getPull() {
115        return pullMixin.getPull();
116    }
117
118    @Override
119    public void setVisibleOn(final String deviceSizeString) {
120        StyleHelper.setVisibleOn(this, deviceSizeString);
121    }
122
123    @Override
124    public void setHiddenOn(final String deviceSizeString) {
125        StyleHelper.setHiddenOn(this, deviceSizeString);
126    }
127
128    /**
129     * Makes this a "drop up" container for dropdown menus where the menu opens upwards.
130     *
131     * @param dropUp
132     */
133    public void setDropUp(final boolean dropUp) {
134        if (dropUp) {
135            addStyleName(Styles.DROP_UP);
136        } else {
137            removeStyleName(Styles.DROP_UP);
138        }
139    }
140
141    @Override
142    public void add(final Widget w) {
143        super.add(w);
144
145        if (name == null) {
146            return;
147        }
148
149        // Add group's name to child widgets that can have a name
150        if (w instanceof HasName) {
151            ((HasName) w).setName(name);
152        }
153    }
154}
