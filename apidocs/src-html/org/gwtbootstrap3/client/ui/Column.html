023import com.google.gwt.dom.client.Style;
024import com.google.gwt.uibinder.client.UiConstructor;
025import com.google.gwt.user.client.ui.Widget;
026import org.gwtbootstrap3.client.ui.constants.ColumnOffset;
027import org.gwtbootstrap3.client.ui.constants.ColumnPull;
028import org.gwtbootstrap3.client.ui.constants.ColumnPush;
029import org.gwtbootstrap3.client.ui.constants.ColumnSize;
030
031/**
032 * A column of Bootstrap's fluid grid system.
033 *
034 * @author Sven Jacobs
035 * @author Joshua Godi
036 * @author Pontus Enmark
037 * @see Row
038 */
039public class Column extends Div {
040
041    private static final String SEPARATOR = "[, ]+";
042
043    public Column(final ColumnSize size, final Widget... widgets) {
044        this(size);
045
046        for (final Widget widget : widgets) {
047            add(widget);
048        }
049    }
050
051    /**
052     * Creates column with primary size.
053     * <p/>
054     * Additional sizes can be added with {@link #addSize(ColumnSize...)}
055     *
056     * @param size Size of column
057     * @see #Column(ColumnSize...)
058     * @see #addSize(ColumnSize...)
059     */
060    public Column(final ColumnSize size) {
061        this();
062        setStyleName(size.getCssName());
063    }
064
065    /**
066     * Creates column with multiple sizes.
067     *
068     * @param sizes Column sizes
069     */
070    public Column(final ColumnSize... sizes) {
071        setSize(sizes);
072    }
073
074    /**
075     * Convenience constructor for UiBinder to create a Column with one or more
076     * sizes.
077     * <p/>
078     * Size needs to be a space-separated String of {@link ColumnSize} enum
079     * names, e.g. "SM_3 LG_3"
080     *
081     * @param size Space-separated String of {@link ColumnSize}
082     * @see ColumnSize
083     */
084    @UiConstructor
085    public Column(final String size) {
086        this();
087        setSize(size);
088    }
089
090    /**
091     * Adds one or more additional column sizes.
092     *
093     * @param sizes Additional column sizes
094     */
095    public void setSize(final ColumnSize... sizes) {
096        addEnumVarargsValues(sizes, ColumnSize.class, true);
097    }
098
099    public void setSize(final String sizes) {
100        addEnumStringValues(sizes, ColumnSize.class, true);
101    }
102
103    public void addSize(final ColumnSize... sizes) {
104        addEnumVarargsValues(sizes, ColumnSize.class, false);
105    }
106
107    public void addSize(final String sizes) {
108        addEnumStringValues(sizes, ColumnSize.class, false);
109    }
110
111    public void setPull(final ColumnPull... pulls) {
112        addEnumVarargsValues(pulls, ColumnPull.class, true);
113    }
114
115    public void setPull(final String pulls) {
116        addEnumStringValues(pulls, ColumnPull.class, true);
117    }
118
119    public void addPull(final ColumnPull... pulls) {
120        addEnumVarargsValues(pulls, ColumnPull.class, false);
121    }
122
123    public void addPull(final String pulls) {
124        addEnumStringValues(pulls, ColumnPull.class, false);
125    }
126
127    public void setPush(final ColumnPush... pushes) {
128        addEnumVarargsValues(pushes, ColumnPush.class, true);
129    }
130
131    public void setPush(final String pushes) {
132        addEnumStringValues(pushes, ColumnPush.class, true);
133    }
134
135    public void addPush(final ColumnPush... pushes) {
136        addEnumVarargsValues(pushes, ColumnPush.class, false);
137    }
138
139    public void addPush(final String pushes) {
140        addEnumStringValues(pushes, ColumnPush.class, false);
141    }
142
143    public void setOffset(final ColumnOffset... offsets) {
144        addEnumVarargsValues(offsets, ColumnOffset.class, true);
145    }
146
147    public void setOffset(final String offsets) {
148        addEnumStringValues(offsets, ColumnOffset.class, true);
149    }
150
151    public void addOffset(final ColumnOffset... offsets) {
152        addEnumVarargsValues(offsets, ColumnOffset.class, false);
153    }
154
155    public void addOffset(final String offsets) {
156        addEnumStringValues(offsets, ColumnOffset.class, false);
157    }
158
159    private <E extends Enum<? extends Style.HasCssName>> void addEnumVarargsValues(final E[] values,
160                                                                                   final Class<E> enumClass, final boolean clearOld) {
161        if (clearOld) {
162            // Remove the previous values
163            removeStyleNames(enumClass);
164        }
165
166        for (final E value : values) {
167            addStyleName(((Style.HasCssName) value).getCssName());
168        }
169    }
170
171    private <E extends Enum<? extends Style.HasCssName>> void addEnumStringValues(final String values,
172                                                                                  final Class<E> enumClass, final boolean clearOld) {
173        if (clearOld) {
174            // Remove the previous values
175            removeStyleNames(enumClass);
176        }
177
178        // Add new ones
179        final String[] valuesSplit = values.split(SEPARATOR);
180        for (final String value : valuesSplit) {
181            for (final E constant : enumClass.getEnumConstants()) {
182                if (value.equalsIgnoreCase(constant.name())) {
183                    addStyleName(((Style.HasCssName) constant).getCssName());
184                }
185            }
186        }
187    }
188
189    private <E extends Enum<? extends Style.HasCssName>> void removeStyleNames(final Class<E> enumClass) {
190        for (final E constant : enumClass.getEnumConstants()) {
191            removeStyleName(((Style.HasCssName) constant).getCssName());
192        }
193    }
194}
