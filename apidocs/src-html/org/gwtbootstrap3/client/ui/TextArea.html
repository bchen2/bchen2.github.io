023import com.google.gwt.dom.client.Document;
024import com.google.gwt.dom.client.Element;
025import com.google.gwt.dom.client.TextAreaElement;
026import com.google.gwt.user.client.ui.RootPanel;
027import com.google.gwt.user.client.ui.Widget;
028import org.gwtbootstrap3.client.ui.base.TextBoxBase;
029import org.gwtbootstrap3.client.ui.constants.Styles;
030
031public class TextArea extends TextBoxBase {
032
033    /**
034     * Creates a TextArea widget that wraps an existing &lt;textarea&gt;
035     * element.
036     * <p/>
037     * This element must already be attached to the document. If the element is
038     * removed from the document, you must call
039     * {@link RootPanel#detachNow(Widget)}.
040     *
041     * @param element the element to be wrapped
042     */
043    public static TextArea wrap(final Element element) {
044        // Assert that the element is attached.
045        assert Document.get().getBody().isOrHasChild(element);
046
047        final TextArea textArea = new TextArea(element);
048
049        // Mark it attached and remember it for cleanup.
050        textArea.onAttach();
051        RootPanel.detachOnWindowClose(textArea);
052
053        return textArea;
054    }
055
056    /**
057     * Creates an empty text area.
058     */
059    public TextArea() {
060        super(Document.get().createTextAreaElement());
061        setStyleName(Styles.FORM_CONTROL);
062    }
063
064    /**
065     * This constructor may be used by subclasses to explicitly use an existing
066     * element. This element must be a &lt;textarea&gt; element.
067     *
068     * @param element the element to be used
069     */
070    protected TextArea(final Element element) {
071        super(element.<Element>cast());
072        TextAreaElement.as(element);
073        element.addClassName(Styles.FORM_CONTROL);
074    }
075
076    /**
077     * Gets the requested width of the text box (this is not an exact value, as
078     * not all characters are created equal).
079     *
080     * @return the requested width, in characters
081     */
082    public int getCharacterWidth() {
083        return getTextAreaElement().getCols();
084    }
085
086    @Override
087    public int getCursorPos() {
088        return getImpl().getTextAreaCursorPos(getElement());
089    }
090
091    @Override
092    public int getSelectionLength() {
093        return getImpl().getTextAreaSelectionLength(getElement());
094    }
095
096    /**
097     * Gets the number of text lines that are visible.
098     *
099     * @return the number of visible lines
100     */
101    public int getVisibleLines() {
102        return getTextAreaElement().getRows();
103    }
104
105    /**
106     * Sets the requested width of the text box (this is not an exact value, as
107     * not all characters are created equal).
108     *
109     * @param width the requested width, in characters
110     */
111    public void setCharacterWidth(final int width) {
112        getTextAreaElement().setCols(width);
113    }
114
115    /**
116     * Sets the number of text lines that are visible.
117     *
118     * @param lines the number of visible lines
119     */
120    public void setVisibleLines(final int lines) {
121        getTextAreaElement().setRows(lines);
122    }
123
124    private TextAreaElement getTextAreaElement() {
125        return getElement().cast();
126    }
127
128    public void clear() {
129        super.setValue(null);
130    }
131}
