023import com.google.gwt.event.shared.HandlerRegistration;
024import com.google.gwt.user.client.Event;
025import org.gwtbootstrap3.client.shared.event.*;
026import org.gwtbootstrap3.client.ui.constants.Styles;
027
028/**
029 * @author Grant Slender
030 */
031public class Collapse extends Div {
032    private static final String TOGGLE = "toggle";
033    private static final String SHOW = "show";
034    private static final String HIDE = "hide";
035
036    private boolean toggle = true;
037
038    public Collapse() {
039        // Set the default styles
040        setStyleName(Styles.COLLAPSE);
041    }
042
043    @Override
044    protected void onLoad() {
045        super.onLoad();
046
047        // Bind jquery events
048        bindJavaScriptEvents(getElement());
049
050        // Configure the collapse
051        collapse(getElement(), toggle);
052    }
053
054    @Override
055    protected void onUnload() {
056        super.onUnload();
057
058        // Unbind the events
059        unbindJavaScriptEvents(getElement());
060    }
061
062    /**
063     * Sets the default state to show or hide. Show is true.
064     */
065    public void setToggle(final boolean toggle) {
066        this.toggle = toggle;
067    }
068
069    /**
070     * Causes the collapse to show or hide
071     */
072    public void toggle() {
073        fireMethod(getElement(), TOGGLE);
074    }
075
076    /**
077     * Causes the collapse to show
078     */
079    public void show() {
080        fireMethod(getElement(), SHOW);
081    }
082
083    /**
084     * Causes the collapse to hide
085     */
086    public void hide() {
087        fireMethod(getElement(), HIDE);
088    }
089
090    public HandlerRegistration addShowHandler(final ShowHandler showHandler) {
091        return addHandler(showHandler, ShowEvent.getType());
092    }
093
094    public HandlerRegistration addShownHandler(final ShownHandler shownHandler) {
095        return addHandler(shownHandler, ShownEvent.getType());
096    }
097
098    public HandlerRegistration addHideHandler(final HideHandler hideHandler) {
099        return addHandler(hideHandler, HideEvent.getType());
100    }
101
102    public HandlerRegistration addHiddenHandler(final HiddenHandler hiddenHandler) {
103        return addHandler(hiddenHandler, HiddenEvent.getType());
104    }
105
106    /**
107     * Fired when the collapse is starting to show
108     */
109    private void onShow(final Event evt) {
110        fireEvent(new ShowEvent(evt));
111    }
112
113    /**
114     * Fired when the collapse has shown
115     */
116    private void onShown(final Event evt) {
117        fireEvent(new ShownEvent(evt));
118    }
119
120    /**
121     * Fired when the collapse is starting to hide
122     */
123    private void onHide(final Event evt) {
124        fireEvent(new HideEvent(evt));
125    }
126
127    /**
128     * Fired when the collapse has hidden
129     */
130    private void onHidden(final Event evt) {
131        fireEvent(new HiddenEvent(evt));
132    }
133
134    private native void bindJavaScriptEvents(final com.google.gwt.dom.client.Element e) /*-{
135        var target = this;
136        var $collapse = $wnd.jQuery(e);
137
138        $collapse.on('show.bs.collapse', function (evt) {
139            target.@org.gwtbootstrap3.client.ui.Collapse::onShow(Lcom/google/gwt/user/client/Event;)(evt);
140        });
141
142        $collapse.on('shown.bs.collapse', function (evt) {
143            target.@org.gwtbootstrap3.client.ui.Collapse::onShown(Lcom/google/gwt/user/client/Event;)(evt);
144        });
145
146        $collapse.on('hide.bs.collapse', function (evt) {
147            target.@org.gwtbootstrap3.client.ui.Collapse::onHide(Lcom/google/gwt/user/client/Event;)(evt);
148        });
149
150        $collapse.on('hidden.bs.collapse', function (evt) {
151            target.@org.gwtbootstrap3.client.ui.Collapse::onHidden(Lcom/google/gwt/user/client/Event;)(evt);
152        });
153    }-*/;
154
155    private native void unbindJavaScriptEvents(final com.google.gwt.dom.client.Element e) /*-{
156        $wnd.jQuery(e).off('show.bs.collapse');
157        $wnd.jQuery(e).off('shown.bs.collapse');
158        $wnd.jQuery(e).off('hide.bs.collapse');
159        $wnd.jQuery(e).off('hidden.bs.collapse');
160    }-*/;
161
162    private native void collapse(final com.google.gwt.dom.client.Element e, final boolean toggle) /*-{
163        $wnd.jQuery(e).collapse({
164            toggle: toggle
165        });
166    }-*/;
167
168    private native void fireMethod(final com.google.gwt.dom.client.Element e, String method) /*-{
169        $wnd.jQuery(e).collapse(method);
170    }-*/;
171
172    private native void fireMethod(final com.google.gwt.dom.client.Element e, int slideNumber) /*-{
173        $wnd.jQuery(e).collapse(slideNumber);
174    }-*/;
175}
